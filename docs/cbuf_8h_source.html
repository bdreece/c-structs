113int cbuf_init(cbuf_t *const cbuf, const size_t element_size,
114 const size_t capacity) {
115 if (!cbuf) return ERR_NULL;
116
117 if (capacity < 1 || element_size < 1) return ERR_INVALID_ARGUMENT;
118
119 cbuf->capacity = capacity;
120 cbuf->element_size = element_size;
121 cbuf->size = 0;
122 cbuf->head = 0;
123 cbuf->tail = 0;
124 cbuf->elements = calloc(capacity, element_size);
125 if (!cbuf->elements) return ERR_FAILURE;
126
127 return ERR_NONE;
128}

144int cbuf_read(cbuf_t *const cbuf, void *element) {
145 if (!cbuf || !cbuf->elements) return ERR_NULL;
146
147 if (cbuf->capacity == 0 || cbuf->size == 0) return ERR_EMPTY;
148
149 if (cbuf->capacity == 1) {
150 STRUCTS_MEMCPY(element, cbuf->elements, cbuf->element_size);
151 return ERR_NONE;
152 } else {
153 STRUCTS_MEMCPY(element, cbuf->elements + cbuf->head * cbuf->element_size,
154 cbuf->element_size);
155 cbuf->head = (cbuf->head + 1) % cbuf->capacity;
156 cbuf->size--;
157 return ERR_NONE;
158 }
159}
