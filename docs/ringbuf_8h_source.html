161int ringbuf_read(ringbuf_t *const ringbuf, void *element) {
162 if (!ringbuf || !ringbuf->elements) return ERR_NULL;
163 if (ringbuf->capacity == 0 || ringbuf->length == 0) return ERR_EMPTY;
164
165 if (ringbuf->capacity == 1) {
166 ASSERT(memcpy(element, ringbuf->elements, ringbuf->size));
167 } else {
168 ASSERT(memcpy(element,
169 ringbuf->elements + ringbuf->head * ringbuf->size,
170 ringbuf->size));
171 ringbuf->head = (ringbuf->head + 1) % ringbuf->capacity;
172 ringbuf->length--;
173 }
174 return ERR_NONE;
175}
