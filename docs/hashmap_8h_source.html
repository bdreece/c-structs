120int hashmap_init(hashmap_t *map, const size_t key_size, const size_t val_size,
121 int (*cmp)(const void *, const void *, size_t),
122 size_t (*hash)(size_t, const void *), size_t initial_breadth,
123 size_t depth) {
124 if (!map) return ERR_NULL;
125
126 if (!cmp || key_size < 1 || val_size < 1 || initial_breadth < 1 || depth < 1)
127 return ERR_INVALID_ARGUMENT;
128
129 int i, ret;
130 // VLA of pointers to maps
131 if ((ret = vla_init(&map->vla, sizeof(map_t *), initial_breadth)) < 0)
132 return ret;
133
134 map->cmp = cmp;
135 map->hash = hash;
136
137 map_t *m;
138 for (i = 0; i < initial_breadth; i++) {
139 m = (map_t *)malloc(sizeof(map_t));
140 if ((ret = map_init(m, key_size, val_size, true, cmp, depth)) < 0)
141 return ret;
142
143 // Copy pointer into vla
144 if ((ret = vla_enq(&map->vla, (const void *)&m)) < 0) return ret;
145 }
146
147 return ERR_NONE;
148}

155 map_t *m;
156
157 for (i = 0; i < n; i++) {
158 // Get map
159 if ((ret = vla_get(&map->vla, i, (void *)&m)) < 0) return ret;

161 // Deinitialize map
162 if ((ret = map_deinit(m)) < 0) return ret;
163

165 free(m);
166 }

172 map->cmp = (int (*)(const void *, const void *, size_t))NULL;
173 map->hash = (size_t(*)(size_t, const void *))NULL;
174
175 return ERR_NONE;
176}
