182static inline pair_t map_create_pair(const map_t *const map,
183 const void *const key,
184 const void *const val) {
185 pair_t p = {.key = malloc(map->key_size), .val = malloc(map->val_size)};
186
187 ASSERT(STRUCTS_MEMCPY(p.key, key, map->key_size) != NULL);
188 ASSERT(STRUCTS_MEMCPY(p.val, val, map->val_size) != NULL);
189
190 return p;
191}

201long umap_search(const map_t *const map, const void *const key) {
202 // Null guard
203 if (!map || !map->vla.elements | !key) return ERR_NULL;
204
205 int ret;
206 size_t i;
207 const pair_t *p;
208
209 // Linear search
210 for (i = 0; i < map_size(map); i++) {
211 // we don't take ownership of p here
212 if ((ret = vla_getp(&map->vla, i, (void **)&p)) < 0) return ret;
213 if (map->cmp(key, p->key, map->key_size) == 0) return i;
214 }
215
216 // key not found
217 return ERR_NOT_FOUND;
218}

220long omap_search(const map_t *const map, const void *const key) {
221 // Null guard
222 if (!map || !map->vla.elements | !key) return ERR_NULL;
223
224 int ret, compare;
225 size_t l, r, m;
226 const pair_t *p;
227
228 l = 0;
229 r = map_size(map);
230
231 // Base case guards (size [0,1])
232 if (r == 0)
233 return ERR_EMPTY;
234 else if (r == 1) {
235 if ((ret = vla_getp(&map->vla, 0, (void **)&p)) < 0) return ret;
236 compare = map->cmp(key, p->key, map->key_size);
237
238 if (compare != 0)
239 return ERR_NOT_FOUND;
240 else
241 return 0;
242 }
243
244 while (r - l > 1) {
245 m = l + (size_t)((r - l) / 2);
246
247 // Not copying p here
248 if ((ret = vla_getp(&map->vla, m, (void **)&p)) < 0) return ret;
249
250 compare = map->cmp(key, p->key, map->key_size);
251
252 if (compare > 0)
253 l = m + 1;
254 else if (compare < 0)
255 r = m - 1;
256 else
257 return m;
258 }
259
260 if ((ret = vla_getp(&map->vla, l, (void **)&p)) < 0) return ret;
261 if (map->cmp(key, p->key, map->key_size) == 0) return l;
262 if (l < r) {
263 if ((ret = vla_getp(&map->vla, r, (void **)&p)) < 0) return ret;
264 if (map->cmp(key, p->key, map->key_size) == 0) return r;
265 }
266
267 return ERR_NOT_FOUND;
268}

270int map_init(map_t *const map, const size_t key_size, const size_t val_size,
271 bool ordered, int (*cmp)(const void *, const void *, size_t),
272 const size_t initial_capacity) {
273 int ret;
274 if (!map || !cmp) return ERR_NULL;
275
276 if (key_size < 1 || val_size < 1 || initial_capacity < 1)
277 return ERR_INVALID_ARGUMENT;
278
279 if ((ret = vla_init(&map->vla, sizeof(pair_t), initial_capacity)) < 0)
280 return ret;
281
282 map->key_size = key_size;
283 map->val_size = val_size;
284 map->cmp = cmp;
285
286 map->search = ordered ? omap_search : umap_search;
287
288 return ERR_NONE;

367int map_ins(map_t *const map, const void *const key, const void *const val) {
368 if (!map || !map->vla.elements || !key || !val) return ERR_NULL;
369
370 int ret;
371 pair_t p;
372 const size_t i = map->search(map, key);
373
374 if (i == ERR_EMPTY) {
375 // Empty map, just enq
376 p = map_create_pair(map, key, val);
377 if ((ret = vla_enq(&map->vla, (void *)&p)) < 0) return ret;
378 } else if (i == ERR_NOT_FOUND) {
379 // Need to search for insertion spot
380 const pair_t *pair;
381 size_t m, l = 0, r = map_size(map);
382 int cmp;
383
384 p = map_create_pair(map, key, val);
385
386 // TODO: shortcut binary search for in-order insertions
387 if ((ret = vla_getp(&map->vla, 0, (void **)&pair)) < 0) return ret;
388 if ((cmp = map->cmp(key, pair->key, map->key_size)) < 0) {
389 if ((ret = vla_ins(&map->vla, 0, (void *)&p)) < 0) return ret;
390 return ERR_NONE;
391 } else {
392 if ((ret = vla_getp(&map->vla, vla_size(&map->vla), (void **)&pair)) < 0)
393 return ret;
394 if ((cmp = map->cmp(key, pair->key, map->key_size)) > 0) {
395 if ((ret = vla_enq(&map->vla, (void *)&p)) < 0) return ret;
396 return ERR_NONE;
397 }
398 }
399
400 while (r - l > 1) {
401 m = l + ((r - l) / 2);
402 if ((ret = vla_getp(&map->vla, m, (void **)&pair)) < 0) return ret;
403 cmp = map->cmp(key, pair->key, map->key_size);
404 if (cmp > 0)
405 l = m;
406 else
407 r = m;
408 }
409
410 // key goes either before l, before r, or after r
411 if ((ret = vla_getp(&map->vla, l, (void **)&pair)) < 0) return ret;
412 cmp = map->cmp(key, pair->key, map->key_size);
413 if (cmp < 0) {
414 // Insert before l
415 if ((ret = vla_ins(&map->vla, l, (void *)&p)) < 0) return ret;
416 } else {
417 if ((ret = vla_getp(&map->vla, r, (void **)&pair)) < 0) return ret;
418 cmp = map->cmp(key, pair->key, map->key_size);
419 if (cmp < 0) {
420 // Insert before r
421 if ((ret = vla_ins(&map->vla, r, (void *)&p)) < 0) return ret;
422 } else {
423 // Insert after r
424 if ((ret = vla_ins(&map->vla, r + 1, (void *)&p)) < 0) return ret;
425 }
426 }
427 } else {
428 // Return error? map_set?
429 if ((ret = map_set(map, key, val)) < 0) return ret;
430 }
431
432 return ERR_NONE;
433}

435int map_del(map_t *const map, const void *const key) {
436 if (!map || !map->vla.elements) return ERR_NULL;
437
438 int ret;
439 pair_t *p;
440 const size_t i = map->search(map, key);
441
442 // Key not found
443 if (i < 0) return i;
444
445 // Key found
446 // vla_getp to access heap pointer, instead of copying
447 if ((ret = vla_getp(&map->vla, i, (void **)&p)) < 0) return ret;
448
449 // Free key and value
450 if ((ret = map_destroy_pair(p)) < 0) return ret;
451
452 // Remove from vla
453 if ((ret = vla_del(&map->vla, i)) < 0) return ret;
454
455 return ERR_NONE;
456}
