204static inline map_pair_t map_create_pair(const map_t *const map,
205 const void *const key,
206 const void *const val) {
207 map_pair_t p = {.key = malloc(map->key_size), .val = malloc(map->val_size)};
208 ASSERT(memcpy(p.key, key, map->key_size));
209 ASSERT(memcpy(p.val, val, map->val_size));
210 return p;
211}

221static struct map_search_result umap_search(const map_t *const map,
222 const void *const key) {
223 struct map_search_result result = {
224 .index = 0,
225 .found = false,
226 };
227
228 const map_pair_t *p;
229
230 // Linear search
231 for (result.index = 0; result.index < map->vla.length; result.index++) {
232 // we don't take ownership of p here
233 ASSERT(vla_getp(&map->vla, result.index, (void **)&p) == ERR_NONE);
234 if (map->cmp(key, p->key) == 0) {
235 result.found = true;
236 return result;
237 }
238 }
239
240 // key not found
241 return result;
242}

244static struct map_search_result omap_search_helper(const map_t *const map,
245 const void *const key,
246 size_t l, size_t r) {
247 struct map_search_result result = {
248 .index = 0,
249 .found = false,
250 };
251
252 int compare;
253 const map_pair_t *p;
254
255 if (map_length(map) == 0) return result;
256
257 if (l >= r) {
258 result.index = r;
259 if (r >= map_length(map)) return result;
260 ASSERT(vla_getp(&map->vla, result.index, (void **)&p) == ERR_NONE);
261 compare = map->cmp(key, p->key);
262 if (compare == 0) {
263 result.found = true;
264 }
265 return result;
266 }
267
268 result.index = l + (r - l) / 2;
269 ASSERT(vla_getp(&map->vla, result.index, (void **)&p) == ERR_NONE);
270 compare = map->cmp(key, p->key);
271
272 switch (compare) {
273 case -1:
274 return omap_search_helper(map, key, l, result.index);
275 case 0:
276 result.found = true;
277 return result;
278 case 1:
279 return omap_search_helper(map, key, result.index + 1, r);
280 }
281}

283static struct map_search_result omap_search(const map_t *const map,

379int map_ins(map_t *const map, const void *const key, const void *const val) {
380 if (!map || !key || !val) return ERR_NULL;
381
382 int ret;
383 map_pair_t p;
384 const struct map_search_result result = map->search(map, key);
385
386 if (!result.found) {
387 p = map_create_pair(map, key, val);
388 ERR_PROP(ret, vla_ins(&map->vla, result.index, (void *)&p));
389 } else {
390 ERR_PROP(ret, vla_get(&map->vla, result.index, (void *)&p));
391 ASSERT(memcpy(p.val, val, map->val_size));
392 }
393 return ERR_NONE;
394}
